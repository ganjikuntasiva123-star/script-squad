<script>

let complaints = {}

function submitComplaint(e){
e.preventDefault()

let id = "CMP" + Math.floor(Math.random()*10000)

complaints[id] = "Submitted"

document.getElementById("msg").innerHTML =
"Complaint submitted successfully. Your ID: <b>"+id+"</b>"

}

function trackComplaint(){

let id = document.getElementById("cid").value

if(complaints[id]){
document.getElementById("status").innerHTML =
"Complaint Status: " + complaints[id]
}
else{
document.getElementById("status").innerHTML =
"Complaint ID not found"
}

}
